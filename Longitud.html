C√≥digo del Ejemplo

frase = "Mis lenguajes favoritos son: html, php, python, pero me quedo con python."
tokens = frase.replace(",", "").replace(":", "").replace(".", "").split()
# ['Mis', 'lenguajes', 'favoritos', 'son', 'html', 'php', 'python', 'pero', 'me', 'quedo', 'con', 'python']

# Crear una lista con las palabras verificadas
verificadas = []  # lista vac√≠a donde se agregar√°n las palabras ya consultadas

for palabra in tokens:
    if palabra not in verificadas:
        verificadas.append(palabra)
        print(palabra, "tiene", len(palabra), "letras")
