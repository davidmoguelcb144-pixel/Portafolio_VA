# Creando listas con las palabras para detectar el sentir de la frase
palabraspositivas = ['feliz', 'contento', 'alegre', 'maravillos', 'bonito', 'bonita']
palabrasnegativas = ['triste', 'solo', 'aburrido', 'malo', 'deprimido', 'horrible', 'muerte']

frase = input("ingresa tu frase")

tokens = frase.replace(",", "").replace(":", "").replace(".", "").split()

positivas = 0
negativas = 0

for palabra in tokens:
    if palabra in palabraspositivas:
        positivas += 1
    elif palabra in palabrasnegativas:
        negativas += 1

print("Palabras positivas:", positivas)
print("Palabras negativas:", negativas)

if positivas > negativas:
    print("La frase es positiva")
elif negativas > positivas:
    print("La frase es negativa")
else:
    print("La frase es neutral")
