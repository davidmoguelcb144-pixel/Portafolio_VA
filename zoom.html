def zoom():
    import cv2

    img_original=cv2.imread('Gatoso.jpg')

    zoom=1.0

    while True:
        img=cv2.resize(img_original,None,fx=zoom, fy=zoom)
        cv2.imshow('Imagen nueva', img)
        tecla=cv2.waitKey(0)
        if tecla==ord('+'):
            zoom=zoom*1.2
        elif tecla==ord('-'):
            zoom=zoom*0.5
        elif tecla == ord('o'):
            zoom=zoom*zoom
        elif tecla==ord('x'):
            break


    cv2.destroyAllWindows()

if __name__ == '__main__':
    zoom()
