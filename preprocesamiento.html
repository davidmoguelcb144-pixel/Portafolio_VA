📝 Código del Ejemplo

import re

import nltk
from nltk import word_tokenize
from textblob.tokenizers import sent_tokenize

#descarga de modelo de tokenizaciòn
#nltk.download("punkt_tab")
#importamos las funciones de la librerìa

#ejemplo de texto a tokenizar
texto=("En la materia de lenguaje natural PNL y visión artificial VA, se desarrolla"
       "un portafolio de practicas como material de apoyo para los siguientes"
       "semestres de la especialidad, incluyendo practicas y glosario de funciones"
       "por cada práctica")

#Como buena practica antes de tokenizar, hay que limpiar el texto
#convertir en minuscula
texto_limpio=texto.lower()

#Quitar saltos de línea y espacios
texto_limpio=texto_limpio.strip()

#Elimina caracteres que no sean letras
texto_limpio=re.sub(r'[^a-záéíóúüñ\s]', '', texto_limpio)

#TOKENIZAR
#token por palabras
palabras=word_tokenize(texto_limpio,language='spanish')
print("token en palabras:",palabras)

#token por oraciones
oraciones=sent_tokenize(texto)
print("Token en oraciones:",oraciones)

print("Texto original:",texto)
print(texto_limpio)
